# 使用 frpc 客户端

  1. 本地目录创建 frpc.ini 文件
# 目录结构如下
├── frpc
│   └── frpc.ini

2
3
  1. 编辑 frpc.ini 文件

    详细操作请参考 frp 官方说明

[common]
server_addr = xxx
server_port = 7000
admin_addr = 127.0.0.1
admin_port = 7400
token = xxx
[ssh]
type = tcp
local_ip = 127.0.0.1
local_port = 22
