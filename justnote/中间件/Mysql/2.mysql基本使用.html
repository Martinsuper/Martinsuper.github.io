# mysql 基本使用

# 1. 用户权限相关

# 1.1 修改密码

数据库环境搭建完成，需要修改默认密码才能创建用户

ALTER USER 'root'@'localhost' IDENTIFIED BY 'password';
1

# 1.2 新建用户

CREATE USER 'username'@'localhost' IDENTIFIED BY 'password';
1
  • username : 用户名
  • password : 登录密码
  • localhost : 指定登录的机器为本机，其他机器无权登录，%可以设置任何ip机器都能登录

# 删除用户

DROP USER 'username'@'host'
1

# 1.3 用户授权

GRANT ALL ON *.* TO 'duan'@'%' WITH GRANT OPTION;
1
GRANT privileges ON databasename.tablename TO 'username'@'host'
flush privileges;
1
2
  • privileges ：用户的操作权限,如SELECT , INSERT , UPDATE，授权全部权限 ALL
  • databasename ：数据库名，指定要授予权限的库，如果要对所有库授予权限用*代替
  • tablename ：表名，指定要授予权限的表，如果要对所有表授予权限用*代替
  • username ：被授予权限的用户名
  • host ：被授予权限的主机，如果要对所有主机授予权限用%代替
  • WITH GRANT OPTION 表示该用户可以将自己的权限给别人，也可不加
  • flush privileges 表示刷新权限相关表

# 撤销权限

REVOKE privileges ON databasename.tablename FROM 'username'@'host';
1

用法类似授权

# 查看权限

SHOW GRANTS;
SHOW GRANTS FOR 'username'@'host'
